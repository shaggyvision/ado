function openPdf(pdfUrl) { // En móviles, abrir directamente en nueva pestaña if (/Mobi|Android/i.test(navigator.userAgent)) { window.open(pdfUrl, '_blank'); return; } // En escritorio, mostrar en lightbox const lb = document.getElementById('lightbox'); const frame = document.getElementById('lightboxFrame'); frame.src = pdfUrl + '#toolbar=0'; lb.style.display = 'flex'; }
